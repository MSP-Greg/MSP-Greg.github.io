def initialize(name, format, cast, ruby_name = nil, converter=nil)
  @name = name
  @format = format
  @cast = cast
  @ruby_name = ruby_name
  @converter = converter
  @value = nil
end

def converted_value
  if @converter
    @converter.call(@value)
  else
    @value
  end
end
