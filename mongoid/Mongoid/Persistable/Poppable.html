def pop(pops)
  prepare_atomic_operation do |ops|
    process_atomic_operations(pops) do |field, value|
      values = send(field)
      value > 0 ? values.pop : values.shift
      ops[atomic_attribute_name(field)] = value
    end
    { "$pop" => ops }
  end
end
