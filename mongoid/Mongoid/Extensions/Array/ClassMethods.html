def __mongoize_fk__(association, object)
  if object.resizable?
    object.blank? ? object : association.convert_to_foreign_key(object)
  else
    object.blank? ? [] : association.convert_to_foreign_key(Array(object))
  end
end

def mongoize(object)
  return if object.nil?
  case object
  when ::Array, ::Set
    object.map(&:mongoize)
  end
end
