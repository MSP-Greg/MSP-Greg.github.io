def evolve(object)
  __evolve__(object) do |obj|
    return if obj.nil?
    case obj
    when ::BigDecimal
      if Mongoid.map_big_decimal_to_decimal128
        BSON::Decimal128.new(obj)
      else
        obj.to_s
      end
    when BSON::Decimal128 then obj
    else
      if obj.numeric?
        if Mongoid.map_big_decimal_to_decimal128
          BSON::Decimal128.new(object.to_s)
        else
          obj.to_s
        end
      else
        obj
      end
    end
  end
end
