def add_key(name, strategy, operator, additional = nil, &block)
  define_method(name) do
    method = "__#{strategy}__".to_sym
    Key.new(self, method, operator, additional, &block)
  end
end

def evolve(object)
  __evolve__(object) do |obj|
    obj.try(:to_sym)
  end
end
