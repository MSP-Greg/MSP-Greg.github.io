def __expr_part__(key, value, negating = false)
  if negating
    { key => { "$#{__regexp?(value) ? "not" : "ne"}" => value }}
  else
    { key => value }
  end
end

def __regexp?(object)
  object.is_a?(Regexp) || object.is_a?(BSON::Regexp::Raw)
end

def evolve(object)
  __evolve__(object) do |obj|
    __regexp?(obj) ? obj : obj.to_s
  end
end
