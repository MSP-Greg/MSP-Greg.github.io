def aggregation(operation)
  return self unless operation
  clone.tap do |query|
    unless aggregating?
      query.pipeline.concat(query.selector.to_pipeline)
      query.pipeline.concat(query.options.to_pipeline)
      query.aggregating = true
    end
    yield(query.pipeline)
  end
end
