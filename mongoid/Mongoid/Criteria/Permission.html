def should_permit?(criteria)
  if criteria.respond_to?(:permitted?)
    return criteria.permitted?
  elsif criteria.respond_to?(:each)
    criteria.each do |criterion|
      return false unless should_permit?(criterion)
    end
  end

  true
end
