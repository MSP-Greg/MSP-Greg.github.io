def preload
  @docs.each do |d|
    set_relation(d, [])
  end

  entries = Hash.new { |hash, key| hash[key] = [] }
  each_loaded_document do |doc|
    fk = doc.send(key)
    entries[fk] << doc
  end

  entries.each do |id, docs|
    set_on_parent(id, docs)
  end
end
