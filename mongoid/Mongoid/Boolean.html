def mongoize(object)
  return if object.nil?

  if object.to_s&.match?(/\A(true|t|yes|y|on|1|1.0)\z/i)
    true
  elsif object.to_s&.match?(/\A(false|f|no|n|off|0|0.0)\z/i)
    false
  end
end
