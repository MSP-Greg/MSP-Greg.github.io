def warning(id, message)
  singleton_class.class_eval do
    define_method("warn_#{id}") do
      unless instance_variable_get("@#{id}")
        Mongoid.logger.warn(message)
        instance_variable_set("@#{id}", true)
      end
    end
  end
end
