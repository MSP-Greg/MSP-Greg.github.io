def new_record=(new_value)
  @new_record ||= false
  if @new_record && !new_value
    @previously_new_record = true
  end
  @new_record = new_value
end

  new_record? &&
    embedded_many? &&
    _parent.persisted? &&
    !_parent.delayed_atomic_sets[atomic_path]

def readonly!
  if Mongoid.legacy_readonly
    Mongoid::Warnings.warn_legacy_readonly
    false
  else
    @readonly = true
  end
end
