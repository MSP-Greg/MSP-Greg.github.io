def <=>(other)
  return super unless other.is_a?(Mongoid::Equality)
  attributes["_id"].to_s <=> other.attributes["_id"].to_s
end

def ==(other)
  self.class == other.class &&
      attributes["_id"] == other.attributes["_id"]
end
