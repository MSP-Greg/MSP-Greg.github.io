def logger
  return @logger if defined?(@logger)
  @logger = rails_logger || default_logger
end

def default_logger
  logger = Logger.new(STDERR)
  logger.level = Mongoid::Config.log_level
  logger
end

def rails_logger
  if defined?(::Rails)
    ::Rails.logger
  else
    nil
  end
end
