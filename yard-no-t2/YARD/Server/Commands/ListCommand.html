def run
  Registry.load_all
  options.update(:objects => run_verifier(Registry.all(:class, :module)))
  list_type = request.path.split('/').last
  meth = "generate_#{list_type}_list"
  tpl = fulldoc_template
  if tpl.respond_to?(meth)
    tpl.send(meth)
    cache(tpl.contents)
  else
    not_found
  end
end
