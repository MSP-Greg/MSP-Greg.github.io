def initialize(orig_exception, msg)
  full_message = "#{msg}.\n" \
                 "The underlying error was #{orig_exception.class}:
                 #{orig_exception.full_message(highlight: false)},
                 with backtrace:\n" \
                 "  #{orig_exception.backtrace.join("\n  ")}\n\n" \
                 "Bundler Error Backtrace:"
  super(full_message)
end
