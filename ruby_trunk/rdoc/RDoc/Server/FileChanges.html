def record_changed(file)
  reload_rbs_signatures_if_needed file
  changed_files << file
end

def reload_rbs_signatures_if_needed(file)
  @reload_rbs_signatures = true if @rdoc.auto_discovered_rbs_signature_file?(file)
end
