def name_location
  args = node.arguments or raise
  first_arg = args.arguments.first or raise

  rbs_location(first_arg.location)
end

def type_fingerprint
  [
    "members/mixin",
    self.class.name,
    module_name.to_s,
    annotation&.type_fingerprint
  ]
end
