  @ivar_name = ivar_name

def map_type_name(&block)
  self.class.new(
    location:,
    prefix_location:,
    ivar_name:,
    ivar_name_location:,
    colon_location:,
    type: type.map_type_name { yield _1 },
    comment_location:
  ) #: self
end

def type_fingerprint
  [
    "annots/instance_variable",
    ivar_name.to_s,
    type.to_s,
    comment_location&.source
  ]
end
