def map_type_name(&block)
  mapped_type_args = type_args.map { |type| type.map_type_name { yield _1 } }

  self.class.new(
    location:,
    prefix_location:,
    type_args: mapped_type_args,
    close_bracket_location:,
    comma_locations:
  ) #: self
end
