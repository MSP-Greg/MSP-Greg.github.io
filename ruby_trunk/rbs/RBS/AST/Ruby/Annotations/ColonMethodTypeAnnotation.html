def map_type_name
  self.class.new(
    location:,
    prefix_location:,
    annotations: annotations,
    method_type: method_type.map_type {|type| type.map_type_name { yield _1 }}
  ) #: self
end

def type_fingerprint
  [
    "annots/colon_method_type",

    method_type.to_s
  ]
