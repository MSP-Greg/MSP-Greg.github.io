def each_decl(&block)
  return enum_for(:each_decl) unless block

  @members.each do |member|
    if member.is_a?(Base)
      yield member
    end
  end
end

def type_fingerprint
  result = [] #: Array[untyped]

  result << "decls/module"
  result << module_name.to_s
  result << members.map { _1.type_fingerprint}

  result
end
