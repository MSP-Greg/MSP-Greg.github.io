def free_variables(set = Set.new)
  set.tap do
    types.each do |type|
      type.free_variables set
    end
  end
end

  else
    enum_for :map_type

def sub(s)
  return self if s.empty?

  self.class.new(types: types.map {|ty| ty.sub(s) },
                 location: location)
end

def to_s(level = 0)
  strs = types.map {|ty| ty.to_s(2) }
  if level > 0
    "(#{strs.join(" & ")})"
  else
    strs.join(" & ")
  end
end
