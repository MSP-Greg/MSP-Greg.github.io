def to_dot
  <<~DOT
    digraph "Prism" {
      node [
        fontname=\"Courier New\"
        shape=plain
        style=filled
        fillcolor=gray95
      ];

      #{nodes.map { |node| node.gsub(/\n/, "\n  ") }.join("\n  ")}
      node [shape=point];
      #{waypoints.join("\n  ")}

      #{edges.join("\n  ")}
    }
  DOT
end
