def to_dot
  if port
    "<tr><td align=\"left\" colspan=\"2\" port=\"#{name}\">#{name}</td></tr>"
  else
    "<tr><td align=\"left\">#{name}</td><td>#{CGI.escapeHTML(value || raise)}</td></tr>"
  end
end
