def to_dot
  dot = <<~DOT
    <table border="0" cellborder="1" cellspacing="0" cellpadding="4">
      <tr><td colspan="2"><b>#{name}</b></td></tr>
  DOT

  if fields.any?
    "#{dot}  #{fields.map(&:to_dot).join("\n  ")}\n</table>"
  else
    "#{dot}</table>"
  end
end
