def compile
  and_node(
    location: node.location,
    left: public_send(read_class, location: node.name_loc, **arguments),
    right: public_send(
      write_class,
      location: node.location,
      **arguments,
      name_loc: node.name_loc,
      value: node.value,
      operator_loc: node.operator_loc
    ),
    operator_loc: node.operator_loc
  )
end
