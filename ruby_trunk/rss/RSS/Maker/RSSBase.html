def initialize(feed_version)
  super(self)

  @feed_version = feed_version
  @version = "1.0"
  @encoding = "UTF-8"
  @standalone = nil
end

  feed = make_feed
  setup_xml_stylesheets(feed)
  setup_elements(feed)
  setup_other_elements(feed)
  feed.validate
  feed
