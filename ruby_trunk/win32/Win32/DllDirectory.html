static VALUE
s_AddDllDirectory(VALUE self, VALUE dir)
{
	void *handle;
	int encidx = rb_enc_find_index("utf-16le");
	StringValueCStr(dir);
	dir = rb_str_export_to_enc(dir, rb_enc_from_index(encidx));
	handle = AddDllDirectory((PCWSTR)RSTRING_PTR(dir));
	return PTR2NUM(handle);
}
