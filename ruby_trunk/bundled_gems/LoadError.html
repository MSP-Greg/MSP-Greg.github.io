def message
  return super unless path

  name = path.tr("/", "-")
  if !defined?(Bundler) && Gem::BUNDLED_GEMS::SINCE[name] && !Gem::BUNDLED_GEMS::WARNED[name]
    warn name + Gem::BUNDLED_GEMS.build_message(name)
  end
  super
end
