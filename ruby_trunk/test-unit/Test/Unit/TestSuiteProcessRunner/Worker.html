def initialize(pid, main_to_worker_output, worker_to_main_input)
  @pid = pid
  @main_to_worker_output = main_to_worker_output
  @worker_to_main_input = worker_to_main_input

def wait
  begin
    Process.waitpid(@pid)
  ensure
    begin
      @main_to_worker_output.close
    ensure
      @worker_to_main_input.close
    end
  end
end
