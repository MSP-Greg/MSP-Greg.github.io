def initialize(*tys)
  types = {}
  tys.each do |ty|
    raise ty.inspect unless ty.is_a?(Type)
    types[ty] = true
  end
  super(types)
end

def show
  Fiber[:show_rec] ||= Set[]
  if Fiber[:show_rec].include?(self)
    "...(recursive)..."
  else
    begin
      Fiber[:show_rec] << self
      @types.empty? ? "untyped" : @types.keys.map {|ty| ty.show }.sort.join(" | ")
    ensure
      Fiber[:show_rec].delete(self) || raise
    end
  end
end
