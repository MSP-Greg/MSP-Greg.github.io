def check_module(genv, mod)
  tae = mod.type_aliases[@name]
  if tae && tae.exist?
    update_type_alias(genv, tae)
    return true
  end
  return false
end

def update_type_alias(genv, tae)
  if tae != @type_alias_entity
    @type_alias_entity = tae
    propagate(genv)
  end
end
