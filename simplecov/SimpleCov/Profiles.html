      Overview

      Profiles are ::SimpleCov configuration procs that can be easily loaded using start :rails and defined using

      SimpleCov.profiles.define :foo do
  # SimpleCov configuration here, same as in  SimpleCov.configure

def define(name, &blk)
  name = name.to_sym
  raise "SimpleCov Profile '#{name}' is already defined" unless self[name].nil?

  self[name] = blk
end

def load(name)
  name = name.to_sym
  raise "Could not find SimpleCov Profile called '#{name}'" unless key?(name)

  SimpleCov.configure(&self[name])
end
