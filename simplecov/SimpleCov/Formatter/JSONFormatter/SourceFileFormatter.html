def format
  if SimpleCov.branch_coverage?
    line_coverage.merge(branch_coverage)
  else
    line_coverage
  end
end

def parse_branch(branch)
  {
    type: branch.type,
    start_line: branch.start_line,
    end_line: branch.end_line,
    coverage: parse_line(branch)
  }
end
