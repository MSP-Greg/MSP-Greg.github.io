def fork_with_simplecov(&block)
  if defined?(SimpleCov) && SimpleCov.running
    fork_without_simplecov do
      SimpleCov.at_fork.call(Process.pid)
      yield if block
    end
  else
    fork_without_simplecov(&block)
  end
end
