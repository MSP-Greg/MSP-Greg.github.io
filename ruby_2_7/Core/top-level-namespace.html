static VALUE
opt_W_getter(ID id, VALUE *variable)
{
    switch (*variable) {
      case Qnil:
	return INT2FIX(0);
      case Qfalse:
	return INT2FIX(1);
      case Qtrue:
	return INT2FIX(2);
      default:
	return Qnil;
    }
}

errat_getter(ID id, VALUE *_)

    }
    else {
	return Qnil;
