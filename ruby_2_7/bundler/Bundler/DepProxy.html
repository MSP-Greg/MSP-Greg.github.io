def ==(other)
  return false if other.class != self.class
  dep == other.dep && __platform == other.__platform
end

def to_s
  s = name.dup
  s << " (#{requirement})" unless requirement == Gem::Requirement.default
  s << " #{__platform}" unless __platform == Gem::Platform::RUBY
  s
end
