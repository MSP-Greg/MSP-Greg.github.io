def self.connect(path, timeout)
  Timeout.timeout(timeout) do
    sock = new(path)
    sock
  end
rescue Timeout::Error
  raise TimeoutError
end

def _read_from_socket(nbytes)
  readpartial(nbytes)
rescue EOFError
  raise Errno::ECONNRESET
end
