def build(attributes = {})
  record = build_record(attributes)
  yield(record) if block_given?
  set_new_record(record)
  record
end

def reader(force_reload = false)
  if force_reload && klass
    klass.uncached { reload }
  elsif !loaded? || stale_target?
    reload
  end

  target
end
