  return false unless namespace

  each_path.first.cbase_type?
end

def each_path(&block)
  return to_enum(__method__) unless block

  descendants = []
  last = self
  loop do
    last = last.children.first
    break if last.nil?

    descendants << last
    break unless last.const_type?
  end
  descendants.reverse_each(&block)

  self
end
