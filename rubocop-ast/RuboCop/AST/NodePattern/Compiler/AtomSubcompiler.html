def visit_other_type
  compiler.with_temp_variables do |compare|
    code = compiler.compile_as_node_pattern(node, var: compare)
    "->(#{compare}) { #{code} }"
  end
end

    raise Invalid, 'unified variables can not appear first as argument'
  end
