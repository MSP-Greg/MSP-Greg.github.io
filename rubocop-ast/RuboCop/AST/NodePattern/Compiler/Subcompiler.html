  us = self
  base.class_eval { @registry = us.registry.dup }
  super
end

def method_added(method)
  @registry[Regexp.last_match(1).to_sym] = method if method =~ /^visit_(.*)/
  super
end

def compile(node)
  prev = @node
  @node = node
  do_compile
ensure
  @node = prev
end

def do_compile
  send(self.class.registry.fetch(node.type, :visit_other_type))
