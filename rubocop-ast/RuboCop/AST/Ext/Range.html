def begins_its_line?
  if (regex = LINE_BEGINS_REGEX_CACHE[column])
    source_line.match?(regex)
  else
    source_line.index(/\S/) == column
  end
end

def same_line?(other)
  other_line = if other.respond_to?(:line)
                 other.line
               elsif other.respond_to?(:loc)
                 other.loc.line
               end

  !other_line.nil? && line == other_line
end
