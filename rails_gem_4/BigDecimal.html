def encode_with(coder)
  string = to_s
  coder.represent_scalar(nil, YAML_MAPPING[string] || string)
end

def to_formatted_s(*args)
  if args[0].is_a?(Symbol)
    super
  else
    format = args[0] || DEFAULT_STRING_FORMAT
    _original_to_s(format)
  end
end
