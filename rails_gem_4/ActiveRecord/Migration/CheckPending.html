def call(env)
  if connection.supports_migrations?
    mtime = ActiveRecord::Migrator.last_migration.mtime.to_i
    if @last_check < mtime
      ActiveRecord::Migration.check_pending!(connection)
      @last_check = mtime
    end
  end
  @app.call(env)
end
