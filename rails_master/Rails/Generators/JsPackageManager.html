def self.detect(root)
  if root.join("bun.lock").exist? || root.join("bun.lockb").exist? || root.join("bun.config.js").exist?
    :bun
  elsif root.join("pnpm-lock.yaml").exist?
    :pnpm
  elsif root.join("package-lock.json").exist?
    :npm
  else
    :yarn
  end
end
