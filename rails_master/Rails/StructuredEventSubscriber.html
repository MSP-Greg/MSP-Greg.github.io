def deprecation(event)
  emit_event("rails.deprecation",
    message: event.payload[:message],
    callstack: event.payload[:callstack],
    gem_name: event.payload[:gem_name],
    deprecation_horizon: event.payload[:deprecation_horizon],
  )
end
