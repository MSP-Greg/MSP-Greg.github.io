def display_name
  base_name = "#{job_data["job_class"]} [#{job_data["job_id"]}] from DelayedJob(#{job_data["queue_name"]})"

  return base_name unless log_arguments?

  "#{base_name} with arguments: #{job_data["arguments"]}"
end
