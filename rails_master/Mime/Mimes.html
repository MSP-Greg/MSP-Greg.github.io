def <<(type)
  @mimes << type
  sym_type = type.to_sym
  @symbols << sym_type
  @symbols_set << sym_type
end

def delete_if
  @mimes.delete_if do |x|
    if yield x
      sym_type = x.to_sym
      @symbols.delete(sym_type)
      @symbols_set.delete(sym_type)
      true
    end
  end
end

def valid_symbols?(symbols) # :nodoc
  symbols.all? { |s| @symbols_set.include?(s) }
