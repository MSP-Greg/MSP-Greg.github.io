def add_bind(bind, &)
  bind = bind.value_for_database if bind.respond_to?(:value_for_database)
  self << quoter.quote(bind)
end

def add_binds(binds, proc_for_binds = nil, &)
  self << binds.map { |bind| quoter.quote(bind) }.join(", ")
end
