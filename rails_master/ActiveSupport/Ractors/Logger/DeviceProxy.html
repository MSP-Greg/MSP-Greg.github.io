  @writer = Writer.spawn(*args, **logdev_options)
  @closed = false

def close
  return true if @closed

  @writer.shutdown
  @closed = true unless frozen?
  true
end

  @writer.reopen(log, options) unless @closed

def write(message)
  return if @closed

  @writer.async(message)
  message.bytesize
end
