def add(words)
  words = words.flatten.map(&:downcase)
  @members.concat(words)
  @pattern = nil
  self
end

def uncountable?(str)
  if @pattern.nil?
    members_pattern = Regexp.union(@members.map { |w| /#{Regexp.escape(w)}/i })
    @pattern = /\b#{members_pattern}\Z/i
  end
  @pattern.match?(str)
end
