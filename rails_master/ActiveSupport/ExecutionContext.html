def set(**options)
  options.symbolize_keys!
  keys = options.keys

  store = record.store

  previous_context = if block_given?
    keys.zip(store.values_at(*keys)).to_h
  end

  store.merge!(options)
  @after_change_callbacks.each(&:call)

  if block_given?
    begin
      yield
    ensure
      store.merge!(previous_context)
      @after_change_callbacks.each(&:call)
    end
  end
end
