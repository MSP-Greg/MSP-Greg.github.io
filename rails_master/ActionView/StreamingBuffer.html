def <<(value)
  value = value.to_s
  value = ERB::Util.h(value) unless value.html_safe?
  @block.call(value)
end

def capture
  buffer = +""
  old_block, @block = @block, ->(value) { buffer << value }
  yield
  buffer.html_safe
ensure
  @block = old_block
end
