def translation
  method_and_value = @tag_value.present? ? "#{@method_name}.#{@tag_value}" : @method_name

  content ||= Translator
    .new(object, @object_name, method_and_value, scope: "helpers.label")
    .translate
  content ||= @method_name.humanize

  content
end
