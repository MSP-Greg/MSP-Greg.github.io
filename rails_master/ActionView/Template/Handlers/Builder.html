def call(template, source)
  require_engine
  # the double assignment is to silence "assigned but unused variable" warnings
  "xml = xml = ::Builder::XmlMarkup.new(indent: 2, target: output_buffer.raw);" \
    "#{source};" \
    "output_buffer.to_s"
end
