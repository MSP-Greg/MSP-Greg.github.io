def _set_detail(key, value) # :doc:
  @details = @details.dup if @digest_cache || @details_key
  @digest_cache = nil
  @details_key = nil
  @details[key] = value
end

def details_key # :nodoc:

def disable_cache
  old_value, @cache = @cache, false
  yield
ensure
  @cache = old_value
end
