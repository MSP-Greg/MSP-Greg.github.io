def cache(key, name, prefix, partial, locals)
  @data[key][name][prefix][partial][locals] ||= canonical_no_templates(yield)
end

def size # :nodoc:
  size = 0
  @data.each_value do |v1|
    v1.each_value do |v2|
      v2.each_value do |v3|
        v3.each_value do |v4|
          size += v4.size
        end
      end
    end
  end

  size + @query_cache.size
end
