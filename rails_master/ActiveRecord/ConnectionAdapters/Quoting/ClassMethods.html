def column_name_matcher
  /
    \A
    (
      (?:
        # table_name.column_name | function(one or no argument)
        ((?:\w\.)?\w | \w+\((?:|\g<2>)\))
      )
      (?:(?:\sAS)?\s\w+)?
    )
    (?:\s*,\s*\g<1>)*
    \z
  /ix
end

def column_name_with_order_matcher
  /
    \A
    (
      (?:
        # table_name.column_name | function(one or no argument)
        ((?:\w\.)?\w | \w+\((?:|\g<2>)\))
      )
      (?:\sASC|\sDESC)?
      (?:\sNULLS\s(?:FIRST|LAST))?
    )
    (?:\s*,\s*\g<1>)*
    \z
  /ix
end
