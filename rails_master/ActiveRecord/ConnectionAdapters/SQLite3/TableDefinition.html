def change_column(column_name, type, **options)
  name = column_name.to_s
  @columns_hash[name] = nil
  column(name, type, **options)
end

def new_column_definition(name, type, **options) # :nodoc:
  case type
  when :virtual
    type = options[:type]
  end

  super
end
