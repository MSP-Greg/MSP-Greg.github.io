def call(*, payload)
  return if payload[:cached]

  sql = payload[:sql]
  @log_all << sql

  unless payload[:name] == "SCHEMA"
    bound_values = (payload[:binds] || []).map do |value|
      value = value.value_for_database if value.respond_to?(:value_for_database)
      value
    end

    @log_full << [sql, bound_values]
  end
end
