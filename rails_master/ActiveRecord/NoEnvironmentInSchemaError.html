def initialize
  msg = "Environment data not found in the schema. To resolve this issue, run: \n\n        bin/rails db:environment:set"
  if defined?(Rails.env)
    super("#{msg} RAILS_ENV=#{::Rails.env}")
  else
    super(msg)
  end
end
