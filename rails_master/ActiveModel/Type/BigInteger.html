      Overview

      Active Model BigInteger Type

      ::ActiveModel::Attribute type for integers that can be serialized to an unlimited number of bytes. This type is registered under the :big_integer key.

      class Person
  include ActiveModel::Attributes

  attribute :id, :big_integer
end

person = Person.new
person.id = "18_000_000_000"

person.id # => 18000000000

      All casting and serialization are performed in the same way as the standard Integer type.

def serialize(value) # :nodoc:
  case value
  when ::Integer
    # noop
  when ::String
    int = value.to_i
    if int.zero? && value != "0"
      return if non_numeric_string?(value)
    end
    value = int
  else
    value = super
  end

  value
end
