def self.for(verbs)
  if verbs.any? { |v| VERB_TO_CLASS[v] == All }
    All
  elsif verbs.one?
    VERB_TO_CLASS[verbs.first]
  else
    Or.new(verbs.map { |v| VERB_TO_CLASS[v] })
  end
end
