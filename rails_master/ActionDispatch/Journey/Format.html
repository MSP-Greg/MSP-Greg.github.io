def initialize(parts)
  @parts      = parts
  @children   = []
  @parameters = []

  parts.each_with_index do |object, i|
    case object
    when Journey::Format
      @children << i
    when Parameter
      @parameters << i
    end
  end
end

def evaluate(hash)
  parts = @parts.dup

  @parameters.each do |index|
    param = parts[index]
    value = hash[param.name]
    return "" if value.nil?
    parts[index] = param.escape value
  end

  @children.each { |index| parts[index] = parts[index].evaluate(hash) }

  parts.join
end
