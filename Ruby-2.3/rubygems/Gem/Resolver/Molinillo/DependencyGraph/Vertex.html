def path_to?(other)
        equal?(other) || successors.any? { |v| v.path_to?(other) }
      end

def recursive_predecessors
        vertices = predecessors
        vertices += vertices.map(&:recursive_predecessors).flatten(1)
        vertices.uniq!
        vertices
      end
