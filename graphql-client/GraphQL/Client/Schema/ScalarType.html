def initialize(type)
  unless type.kind.scalar?
    raise "expected type to be a Scalar, but was #{type.class}"
  end

  @type = type
end

def cast(value, _errors = nil)
  case value
  when NilClass
    nil
  else
    if type.respond_to?(:coerce_isolated_input)
      type.coerce_isolated_input(value)
    else
      type.coerce_input(value)
    end
  end
end
