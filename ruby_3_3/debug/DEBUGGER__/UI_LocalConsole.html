def activate_sigint
  prev_handler = trap(:SIGINT){
    if SESSION.active?
      ThreadClient.current.on_trap :SIGINT
    end
  }
  SESSION.intercept_trap_sigint_start prev_handler
end

def after_fork_parent
  parent_pid = Process.pid

  at_exit{
    SESSION.intercept_trap_sigint_end
    trap(:SIGINT, :IGNORE)

    if Process.pid == parent_pid
      # only check child process from its parent
      begin
        # wait for all child processes to keep terminal
        Process.waitpid
      rescue Errno::ESRCH, Errno::ECHILD
      end
    end
  }
end

def deactivate_sigint
  if SESSION.intercept_trap_sigint?
    prev = SESSION.intercept_trap_sigint_end
    trap(:SIGINT, prev)
  end
end

def puts str = nil
  case str
  when Array
    str.each{|line|
      $stdout.puts line.chomp
    }
  when String
    str.each_line{|line|
      $stdout.puts line.chomp
    }
  when nil
    $stdout.puts
  end
end

def setup_interrupt
  SESSION.intercept_trap_sigint false do
    current_thread = Thread.current # should be session_server thread

    prev_handler = trap(:INT){
      current_thread.raise Interrupt
    }

    yield
  ensure
    trap(:INT, prev_handler)
  end
end
