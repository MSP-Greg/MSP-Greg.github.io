def self.with(&block)
  pointer = FFI::MemoryPointer.new(SIZEOF)

  begin
    raise unless LibRubyParser.pm_buffer_init(pointer)
    yield new(pointer)
  ensure
    LibRubyParser.pm_buffer_free(pointer)
    pointer.free
  end
end
