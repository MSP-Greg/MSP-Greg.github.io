def filter bt
  return ["No backtrace"] unless bt

  return bt.dup if $DEBUG || ENV["MT_DEBUG"]

  new_bt = bt.take_while { |line| line !~ MT_RE }
  new_bt = bt.select     { |line| line !~ MT_RE } if new_bt.empty?
  new_bt = bt.dup                                 if new_bt.empty?

  new_bt
end
