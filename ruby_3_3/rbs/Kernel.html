def TypeName(string)
  absolute = string.start_with?("::")

  *path, name = string.delete_prefix("::").split("::").map(&:to_sym)
  raise unless name

  RBS::TypeName.new(
    name: name,
    namespace: RBS::Namespace.new(path: path, absolute: absolute)
  )
end
