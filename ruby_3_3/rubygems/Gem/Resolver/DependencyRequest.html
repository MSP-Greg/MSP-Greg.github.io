      Overview

def ==(other) # :nodoc:
  case other
  when Gem::Dependency
    @dependency == other
  when Gem::Resolver::DependencyRequest
    @dependency == other.dependency
  else
    false
  end
end

def pretty_print(q) # :nodoc:
  q.group 2, "[Dependency request ", "]" do
    q.breakable
    q.text @dependency.to_s

    q.breakable
    q.text " requested by "
    q.pp @requester
  end
end
