def initialize(dir)
  files = Dir.glob(File.join(dir, "**", "*")).map {|f| "- #{f}" }.join "\n"

  super <<~MSG
    Dynamic library not found for Rust extension (in #{dir})

    Make sure you set "crate-type" in Cargo.toml to "cdylib"

    Found files:
    #{files}
  MSG
end
