    tbl[v] = true
  end
  new(tbl)

def intersection(other)
  tbl = {}
  h = 0
  each do |elem|
    if other.include?(elem)
      tbl << elem
      h ^= elem.hash
    end
  end
  Set.new(tbl, h)
end

def map(&blk)
  tbl = {}
  each do |elem|
    v = yield(elem)
    tbl[v] = true
  end
  Set.new(tbl)
end

def sum(other)
  if @tbl.size == 0
    other
  elsif other.tbl.size == 0
    self
  else
    Set.new(@tbl.merge(other.tbl))
  end
end
