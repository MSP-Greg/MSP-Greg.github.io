def self.from_rbs(rbs_loc)
  CodeRange.new(
    CodeLocation.new(rbs_loc.start_line, rbs_loc.start_column),
    CodeLocation.new(rbs_loc.end_line, rbs_loc.end_column),
  )
end

def overlap?(other)
  if @first <= other.first
    return @last > other.first
  else
    return @first < other.last
  end
end
