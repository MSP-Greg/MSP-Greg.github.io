def handle_pended_error(exception)
  return false unless exception.is_a?(PendedError)
  pending = Pending.new(name,
                        filter_backtrace(exception.backtrace),
                        exception.message,
                        :method_name => @method_name)
  add_pending(pending)
  true
end
