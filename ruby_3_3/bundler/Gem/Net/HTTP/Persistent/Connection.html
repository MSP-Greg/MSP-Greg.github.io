def initialize http_class, http_args, ssl_generation
  @http           = http_class.new(*http_args)
  @ssl_generation = ssl_generation

  reset
end

def finish
  @http.finish
rescue IOError
ensure
  reset
end

def reset
  @last_use = Gem::Net::HTTP::Persistent::EPOCH
  @requests = 0
end
