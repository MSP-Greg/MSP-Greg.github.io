def self.ci_strings
  matching_names = ENV_DESCRIPTORS.select {|env, _| ENV[env] }.values
  matching_names << ENV["CI_NAME"].downcase if ENV["CI_NAME"]
  matching_names.reject(&:empty?).sort.uniq
end
