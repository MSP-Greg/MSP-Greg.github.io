private def sql_error
  return nil unless @sql
  return @sql.chomp unless @sql_offset >= 0

  offset = @sql_offset
  sql.lines.flat_map do |line|
    if offset >= 0 && line.length > offset
      blanks = " " * offset
      offset = -1
      [line.chomp, blanks + "^"]
    else
      offset -= line.length if offset
      line.chomp
    end
  end.join("\n")
end
