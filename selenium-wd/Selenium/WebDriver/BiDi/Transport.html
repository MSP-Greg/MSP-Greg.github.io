def execute(cmd:, params: nil, result: nil)
  reply = @connection.send_cmd(method: cmd, params: serialize(params))
  raise Error::WebDriverError, error_message(reply) if reply['error']

  value = reply['result']
  result ? result.from_json(value) : value
end
