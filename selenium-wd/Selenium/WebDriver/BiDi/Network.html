def continue_with_auth(request_id, username, password)
  @bidi.send_cmd(
    'network.continueWithAuth',
    'request' => request_id,
    'action' => 'provideCredentials',
    'credentials' => {
      'type' => 'password',
      'username' => username,
      'password' => password
    }
  )
end

def on(event, &)
  event = EVENTS[event] if event.is_a?(Symbol)
  @bidi.add_callback(event, &)
end
