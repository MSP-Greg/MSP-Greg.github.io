def initialize(source, duration, x, y, **opts)
  super(source)
  @duration = duration * 1000
  @x_offset = x
  @y_offset = y
  @origin = opts.delete(:element) || opts.delete(:origin) || :viewport
  @type = :pointerMove
  @opts = opts
end

def encode
  process_opts.merge('type' => type.to_s,
                     'duration' => @duration.to_i,
                     'x' => @x_offset,
                     'y' => @y_offset,
                     'origin' => @origin)
end
