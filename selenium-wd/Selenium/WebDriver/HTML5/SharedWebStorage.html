def each
  return enum_for(:each) unless block_given?

  keys.each do |k|
    yield k, self[k]
  end
end

def fetch(key)
  return self[key] if key? key
  return yield(key) if block_given?

  raise KeyError, "missing key #{key.inspect}"
end
