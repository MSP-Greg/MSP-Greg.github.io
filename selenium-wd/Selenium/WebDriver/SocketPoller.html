def listening?
  TCPSocket.new(@host, @port).close
  true
rescue *NOT_CONNECTED_ERRORS
  false
end

def with_timeout
  max_time = current_time + @timeout

  until current_time > max_time
    return true if yield

    sleep @interval
  end

  false
end
