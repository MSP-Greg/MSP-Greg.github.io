def rubocop_version_with_support
  if supported?
    RuboCop::Version::STRING
  else
    OBSOLETE_RUBIES[version]
  end
end

def source
  @source ||= SOURCES.each.lazy.map { |c| c.new(@config) }.detect(&:version)
end
