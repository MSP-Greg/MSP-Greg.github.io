def print_documentation_url
  puts Cop::Documentation.default_base_url if cops_array.empty?

  cops_array.each do |cop_name|
    cop = registry_hash[cop_name]

    next if cop.empty?

    puts Cop::Documentation.url_for(cop.first, @config)
  end

  puts
end
