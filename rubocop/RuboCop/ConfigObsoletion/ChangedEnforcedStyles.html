def message
  base = format(BASE_MESSAGE,
                parameter: parameter, value: value, cop: cop, path: smart_loaded_path)

  if alternative
    "#{base}\n`#{parameter}: #{value}` has been renamed to " \
      "`#{parameter}: #{alternative.chomp}`."
  else
    "#{base}\n#{reason.chomp}"
  end
end
