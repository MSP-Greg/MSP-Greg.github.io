def run(path, contents, options, prism_result: nil)
  @options = options.merge(DEFAULT_RUBOCOP_OPTIONS)
  @options[:stdin] = contents

  @prism_result = prism_result

  @offenses = []
  @warnings = []
  @errors = []

  super([path])

  raise Interrupt if aborting?
end
