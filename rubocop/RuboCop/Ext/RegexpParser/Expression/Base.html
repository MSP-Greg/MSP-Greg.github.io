def build_location
  return { body: expression } unless (q = quantifier)

  body = expression.adjust(end_pos: -q.text.length)
  q.origin = origin
  q_loc = q.expression

  { body: body, quantifier: q_loc }
end

def expression
  @expression ||= origin.adjust(begin_pos: ts, end_pos: ts + full_length)
end
