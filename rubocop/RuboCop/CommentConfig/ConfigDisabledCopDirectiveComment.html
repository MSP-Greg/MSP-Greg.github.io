def initialize(cop_name)
  @text = "# rubocop:disable #{cop_name}"
  @line_number = CONFIG_DISABLED_LINE_RANGE_MIN
  @loc = Loc.new(Expression.new(CONFIG_DISABLED_LINE_RANGE_MIN))
end
