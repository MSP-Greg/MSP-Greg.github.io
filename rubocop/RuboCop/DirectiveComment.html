def cops
  match = comment.text.match(CommentConfig::COMMENT_DIRECTIVE_REGEXP)
  return unless match

  cops_string = match.captures[1]
  cops_string.split(/,\s*/).uniq.sort
end

def match?(cop_names)
  cops == cop_names.uniq.sort
end
