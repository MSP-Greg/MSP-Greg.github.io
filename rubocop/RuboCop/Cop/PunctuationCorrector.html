def add_space(corrector, token)
  corrector.replace(token.pos, "#{token.pos.source} ")
end

def swap_comma(corrector, range)
  return unless range

  case range.source
  when ',' then corrector.remove(range)
  else          corrector.insert_after(range, ',')
  end
end
