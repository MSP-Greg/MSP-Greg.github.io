def mixed_formats?
  formats = format_sequences.reject(&:percent?).map do |seq|
    if seq.name
      :named
    elsif seq.max_digit_dollar_num
      :numbered
    else
      :unnumbered
    end
  end

  formats.uniq.size > 1
end

def parse
  matches = []
  @source.scan(SEQUENCE) { matches << FormatSequence.new(Regexp.last_match) }
  matches
end
