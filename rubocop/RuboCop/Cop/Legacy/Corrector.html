def initialize(source, corr = [])
  super(source)
  if corr.is_a?(CorrectionsProxy)
    merge!(corr.send(:corrector))
  else
    # warn "Corrector.new with corrections is deprecated." unless corr.empty? TODO
    corr.each { |c| corrections << c }
  end
end
