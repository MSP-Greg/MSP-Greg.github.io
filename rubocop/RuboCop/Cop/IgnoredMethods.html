      Overview

      This module encapsulates the ability to ignore certain methods when parsing. Cops that use IgnoredMethods can accept either strings or regexes to match against.

def deprecated_key
  return unless self.class.respond_to?(:deprecated_key)

  self.class.deprecated_key&.to_s
end

def ignored_method?(name)
  ignored_methods.any? do |value|
    case value
    when Regexp
      value.match? String(name)
    else
      value == String(name)
    end
  end
end

def ignored_methods
  keys = %w[IgnoredMethods]
  keys << deprecated_key if deprecated_key

  cop_config.slice(*keys).values.reduce(&:concat)
end
