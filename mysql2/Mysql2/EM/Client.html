def query(sql, opts = {})
  if ::EM.reactor_running?
    super(sql, opts.merge(async: true))
    deferable = ::EM::DefaultDeferrable.new
    @watch = ::EM.watch(socket, Watcher, self, deferable)
    @watch.notify_readable = true
    deferable
  else
    super(sql, opts)
  end
end
