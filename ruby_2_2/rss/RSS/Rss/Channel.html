def setup_maker_elements(channel)
  super
  [
    [skipDays, "day"],
    [skipHours, "hour"],
  ].each do |skip, key|
    if skip
      skip.__send__("#{key}s").each do |val|
        target_skips = channel.__send__("skip#{key.capitalize}s")
        new_target = target_skips.__send__("new_#{key}")
        new_target.content = val.content
      end
    end
  end
end
