def self.new
  path = nil

  # workaround File#path being tainted:
  # https://bugs.ruby-lang.org/issues/14485
  fp = begin
    path = "#{Dir::tmpdir}/#{rand}"
    super(path, RDWR|CREAT|EXCL, 0600)
  rescue Errno::EEXIST
    retry
  end

  unlink(path)
  fp.binmode
  fp.sync = true
  fp
end
