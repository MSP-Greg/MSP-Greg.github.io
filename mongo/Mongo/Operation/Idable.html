def ensure_ids(documents)
  @ids = []
  documents.collect do |doc|
    doc_with_id = has_id?(doc) ? doc : doc.merge(_id: id_generator.generate)
    @ids << id(doc_with_id)
    doc_with_id
  end
end

def id(doc)
  doc.respond_to?(:id) ? doc.id : (doc['_id'] || doc[:_id])
end
