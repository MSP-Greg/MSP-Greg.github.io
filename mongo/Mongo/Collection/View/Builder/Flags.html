def map_flags(options)
  MAPPINGS.each.reduce(options[:flags] || []) do |flags, (key, value)|
    cursor_type = options[:cursor_type]
    if options[key] || (cursor_type && cursor_type == key)
      flags.push(*value)
    end
    flags
  end
end
