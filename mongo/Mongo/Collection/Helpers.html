def do_drop(operation, session, context)
  operation.execute(next_primary(nil, session), context: context)
rescue Error::OperationFailure::Family => ex
  # NamespaceNotFound
  if ex.code == 26 || ex.code.nil? && ex.message =~ /ns not found/
    false
  else
    raise
  end
end
