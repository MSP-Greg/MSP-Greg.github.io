def self.hint_error(**options)
  unacknowledged_write = options[:unacknowledged_write] || false

  error_message = if unacknowledged_write
    UNACKNOWLEDGED_HINT_MESSAGE
  else
    HINT_MESSAGE
  end

  new(error_message)
end
