def write_retryable?
  write_retryable_code? ||
  code.nil? && WRITE_RETRY_MESSAGES.any? { |m| message.include?(m) }

private def write_retryable_code?
  if code
    WRITE_RETRY_ERRORS.any? { |e| e[:code] == code }
  else
    # return false rather than nil
    false
  end
end
