def self.serialize(buffer, value, validating_keys = nil)
  if value.is_a?(BSON::Int32)
    if value.respond_to?(:value)
      # bson-ruby >= 4.6.0
      value = value.value
    else
      value = value.instance_variable_get('@integer')
    end
  end
  buffer.put_int32(value)
end
