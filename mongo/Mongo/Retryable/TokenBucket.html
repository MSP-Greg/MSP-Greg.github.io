      Overview

      A thread-safe token bucket for rate limiting retries during server overload. Used by the adaptive retry mechanism.

def consume(count = 1)
  @mutex.synchronize do
    if @tokens >= count
      @tokens -= count
      true
    else
      false
    end
  end
end

def deposit(count)
  @mutex.synchronize do
    @tokens = [ @capacity, @tokens + count ].min
  end
end
