def to_document
  return BSON::Document.new({}) if @empty
  BSON::Document.new({
    provider: 'gcp',
    projectId: project_id,
    location: location,
    keyRing: key_ring,
    keyName: key_name
  }).tap do |bson|
    unless key_version.nil?
      bson.update({ keyVersion: key_version })
    end
    unless endpoint.nil?
      bson.update({ endpoint: endpoint })
    end
  end
end
