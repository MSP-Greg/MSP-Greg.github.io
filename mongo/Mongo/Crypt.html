def validate_ffi!
  return if defined?(FFI)

  require 'ffi'
rescue LoadError => e
  raise Error::UnmetDependency, 'Cannot enable encryption because the ffi gem ' \
                                "has not been installed. Add \"gem 'ffi'\" to your Gemfile and run " \
                                "\"bundle install\" to install the gem. (#{e.class}: #{e})"
end
