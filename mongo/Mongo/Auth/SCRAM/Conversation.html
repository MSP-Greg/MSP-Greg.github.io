def hi(data)
  OpenSSL::PKCS5.pbkdf2_hmac_sha1(
    data,
    salt,
    iterations,
    digest.size,
  )
end

def salted_password
  @salted_password ||= CredentialCache.cache(cache_key(:salted_password)) do
    hi(user.hashed_password)
  end
end
