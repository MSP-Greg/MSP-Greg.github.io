def get_day(offset, year)
  next_month = month + 1
  if next_month == 13
    year += 1
    next_month = 1
  end

  candidate = Time.new(year, next_month, 1, 0, 0, 0, offset.observed_utc_offset) - 86400
  diff = candidate.wday - day_of_week

  if diff < 0
    candidate - (diff + 7) * 86400
  elsif diff > 0
    candidate - diff * 86400
  else
    candidate
  end
end
