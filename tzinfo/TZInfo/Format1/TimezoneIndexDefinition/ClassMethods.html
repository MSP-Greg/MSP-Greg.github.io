def data_timezones
  unless @data_timezones.frozen?
    @data_timezones = @data_timezones.sort.freeze
  end
  @data_timezones
end

def linked_timezone(identifier)
  identifier = StringDeduper.global.dedupe(identifier)
  @timezones << identifier
  @linked_timezones << identifier
end
