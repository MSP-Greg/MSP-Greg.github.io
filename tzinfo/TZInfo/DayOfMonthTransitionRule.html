def get_day(offset, year)
  candidate = Time.new(year, month, @offset_start, 0, 0, 0, offset.observed_utc_offset)
  diff = day_of_week - candidate.wday

  if diff < 0
    candidate + (7 + diff) * 86400
  elsif diff > 0
    candidate + diff * 86400
  else
    candidate
  end
end
