def c(shape, value, instance = nil)

    converter.call(value, instance)
  else
    value
  end
end

def ensure_open(file, converter)
  if file.closed?
    new_file = File.open(file.path, 'rb')
    converter.opened_files << new_file
    new_file
  else
    file
  end
end
