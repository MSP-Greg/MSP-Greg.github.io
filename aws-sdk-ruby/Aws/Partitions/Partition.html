def region(region_name)
  if @regions.key?(region_name)
    @regions[region_name]
  else
    msg = "invalid region name #{region_name.inspect}; valid region "
    msg << "names include %s" % [@regions.keys.join(', ')]
    raise ArgumentError, msg
  end
end
