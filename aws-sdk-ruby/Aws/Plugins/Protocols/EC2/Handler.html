def parse_xml(context)
  if rules = context.operation.output
    parser = Xml::Parser.new(rules)
    data = parser.parse(xml(context)) do |path, value|
      if path.size == 2 && path.last == 'requestId'
        context.metadata[:request_id] = value
      end
    end
    data
  else
    EmptyStructure.new
  end
end
