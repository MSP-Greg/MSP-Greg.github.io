def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 180,
    delay: 120,
    poller: Aws::Waiters::Poller.new(
      operation_name: :describe_training_job,
      acceptors: [
        {
          "expected" => "Completed",
          "matcher" => "path",
          "state" => "success",
          "argument" => "training_job_status"
        },
        {
          "expected" => "Stopped",
          "matcher" => "path",
          "state" => "success",
          "argument" => "training_job_status"
        },
        {
          "expected" => "Failed",
          "matcher" => "path",
          "state" => "failure",
          "argument" => "training_job_status"
        },
        {
          "expected" => "ValidationException",
          "matcher" => "error",
          "state" => "failure"
        }
      ]
    )
  }.merge(options))
end
