def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 20,
    delay: 5,
    poller: Aws::Waiters::Poller.new(
      operation_name: :describe_multiplex,
      acceptors: [
        {
          "state" => "success",
          "matcher" => "path",
          "argument" => "state",
          "expected" => "DELETED"
        },
        {
          "state" => "retry",
          "matcher" => "path",
          "argument" => "state",
          "expected" => "DELETING"
        },
        {
          "state" => "retry",
          "matcher" => "status",
          "expected" => 500
        }
      ]
    )
  }.merge(options))
end
