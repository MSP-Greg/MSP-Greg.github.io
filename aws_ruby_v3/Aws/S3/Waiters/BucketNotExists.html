def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 20,
    delay: 5,
    poller: Aws::Waiters::Poller.new(
      operation_name: :head_bucket,
      acceptors: [{
        "expected" => 404,
        "matcher" => "status",
        "state" => "success"
      }]
    )
  }.merge(options))
end
