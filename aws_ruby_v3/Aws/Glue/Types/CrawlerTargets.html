      Note:

      When making an API call, you may pass CrawlerTargets data as a hash:

      {
  s3_targets: [
    {
      path: "Path",
      exclusions: ["Path"],
      connection_name: "ConnectionName",
    },
  ],
  jdbc_targets: [
    {
      connection_name: "ConnectionName",
      path: "Path",
      exclusions: ["Path"],
    },
  ],
  mongo_db_targets: [
    {
      connection_name: "ConnectionName",
      path: "Path",
      scan_all: false,
    },
  ],
  dynamo_db_targets: [
    {
      path: "Path",
      scan_all: false,
      scan_rate: 1.0,
    },
  ],
  catalog_targets: [
    {
      database_name: "NameString", # required
      tables: ["NameString"], # required
    },
  ],
}
