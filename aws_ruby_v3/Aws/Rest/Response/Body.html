def apply(body, data)
  if event_stream?
    data[@rules[:payload]] = parse_eventstream(body)
  elsif streaming?
    data[@rules[:payload]] = body
  elsif @rules[:payload]
    data[@rules[:payload]] = parse(body.read, @rules[:payload_member])
  elsif !@rules.shape.member_names.empty?
    parse(body.read, @rules, data)
  end
end
