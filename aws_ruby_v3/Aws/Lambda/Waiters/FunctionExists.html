def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 20,
    delay: 1,
    poller: Aws::Waiters::Poller.new(
      operation_name: :get_function,
      acceptors: [
        {
          "state" => "success",
          "matcher" => "status",
          "expected" => 200
        },
        {
          "state" => "retry",
          "matcher" => "error",
          "expected" => "ResourceNotFoundException"
        }
      ]
    )
  }.merge(options))
end

def wait(params = {})
