def data(bytes, options = {})
  if options[:end_stream]
    @state = :closed
  else
    decoder = Aws::EventStream::Decoder.new
    event = decoder.decode_chunk(bytes).first
    @send_events << decoder.decode_chunk(event.payload.read).first
  end
end
