def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 20,
    delay: 20,
    poller: Aws::Waiters::Poller.new(
      operation_name: :describe_environments,
      acceptors: [
        {
          "state" => "success",
          "matcher" => "pathAll",
          "argument" => "environments[].status",
          "expected" => "Ready"
        },
        {
          "state" => "retry",
          "matcher" => "pathAll",
          "argument" => "environments[].status",
          "expected" => "Updating"
        }
      ]
    )
  }.merge(options))
end
