def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 30,
    delay: 60,
    poller: Aws::Waiters::Poller.new(
      operation_name: :describe_clusters,
      acceptors: [
        {
          "expected" => "available",
          "matcher" => "pathAll",
          "state" => "success",
          "argument" => "clusters[].cluster_status"
        },
        {
          "expected" => "deleting",
          "matcher" => "pathAny",
          "state" => "failure",
          "argument" => "clusters[].cluster_status"
        },
        {
          "expected" => "ClusterNotFound",
          "matcher" => "error",
          "state" => "retry"
        }
      ]
    )
  }.merge(options))
end
