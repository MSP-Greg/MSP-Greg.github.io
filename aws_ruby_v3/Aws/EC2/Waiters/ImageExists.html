def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 40,
    delay: 15,
    poller: Aws::Waiters::Poller.new(
      operation_name: :describe_images,
      acceptors: [
        {
          "matcher" => "path",
          "expected" => true,
          "argument" => "length(images[]) > `0`",
          "state" => "success"
        },
        {
          "matcher" => "error",
          "expected" => "InvalidAMIID.NotFound",
          "state" => "retry"
        }
      ]
    )
  }.merge(options))
end
