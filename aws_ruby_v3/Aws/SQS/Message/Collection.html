def batch_delete!(options = {})
  batch_enum.each do |batch|
    params = Aws::Util.copy_hash(options)
    params[:queue_url] = batch[0].queue_url
    params[:entries] ||= []
    batch.each do |item|
      params[:entries] << {
        id: item.message_id,
        receipt_handle: item.receipt_handle
      }
    end
    batch[0].client.delete_message_batch(params)
  end
  nil
end
