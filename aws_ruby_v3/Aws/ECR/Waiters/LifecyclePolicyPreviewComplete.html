def initialize(options)
  @client = options.fetch(:client)
  @waiter = Aws::Waiters::Waiter.new({
    max_attempts: 20,
    delay: 5,
    poller: Aws::Waiters::Poller.new(
      operation_name: :get_lifecycle_policy_preview,
      acceptors: [
        {
          "state" => "success",
          "matcher" => "path",
          "argument" => "status",
          "expected" => "COMPLETE"
        },
        {
          "state" => "failure",
          "matcher" => "path",
          "argument" => "status",
          "expected" => "FAILED"
        }
      ]
    )
  }.merge(options))
end
