def authorizer(name)
  if @authorizers.key?(name.to_sym)
    @authorizers[name.to_sym]
  else
    raise ArgumentError, "unknown authorizer #{name.inspect}"
  end
end

def authorizers(&block)
  if block_given?
    @authorizers.each(&block)
  else
    @authorizers.enum_for(:each)
  end
end
