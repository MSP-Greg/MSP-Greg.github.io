def drop
  require 'pathname'
  path = Pathname.new configuration['database']
  file = path.absolute? ? path.to_s : File.join(root, path)

  FileUtils.rm(file)
rescue Errno::ENOENT => error
  raise NoDatabaseError.new(error.message, error)
end
