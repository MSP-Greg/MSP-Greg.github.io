def calculate(operation, _column_name)
  if [:count, :sum, :size].include? operation
    group_values.any? ? Hash.new : 0
  elsif [:average, :minimum, :maximum].include?(operation) && group_values.any?
    Hash.new
  else
    nil
  end
end
