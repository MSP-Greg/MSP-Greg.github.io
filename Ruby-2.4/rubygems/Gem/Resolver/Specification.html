      A Specification contains a subset of the information contained in a Specification. Only the information necessary for dependency resolution in the resolver is included.

def install options = {}
  require 'rubygems/installer'

  destination = options[:install_dir] || Gem.dir

  Gem.ensure_gem_subdirectories destination

  gem = source.download spec, destination

  installer = Gem::Installer.at gem, options

  yield installer if block_given?

  @spec = installer.install
end
