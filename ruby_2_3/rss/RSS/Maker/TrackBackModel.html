def self.append_features(klass)
  super

  klass.def_other_element("#{RSS::TRACKBACK_PREFIX}_ping")
  klass.def_classed_elements("#{RSS::TRACKBACK_PREFIX}_about", "value",
                             "TrackBackAbouts")
end
