def column_spec_for_primary_key(column)
  spec = super
  if schema_type(column) == :uuid
    spec[:default] ||= "nil"
  end
  spec
end

def schema_type(column)
  return super unless column.serial?

  if column.bigint?
    :bigserial
  else
    :serial
  end
end
