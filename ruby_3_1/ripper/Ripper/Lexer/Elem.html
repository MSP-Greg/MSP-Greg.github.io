def pretty_print(q)
  q.group(2, "#<#{self.class}:", ">") {
    q.breakable
    q.text("#{event}@#{pos[0]}:#{pos[1]}")
    q.breakable
    state.pretty_print(q)
    q.breakable
    q.text("token: ")
    tok.pretty_print(q)
    if message
      q.breakable
      q.text("message: ")
      q.text(message)
    end
  }
end

def to_a
  if @message
    [@pos, @event, @tok, @state, @message]
  else
    [@pos, @event, @tok, @state]
  end
end
