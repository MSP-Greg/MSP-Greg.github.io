def _parse
  begin
    REXML::Document.parse_stream(@rss, @listener)
  rescue RuntimeError => e
    raise NotWellFormedError.new{e.message}
  rescue REXML::ParseException => e
    context = e.context
    line = context[0] if context
    raise NotWellFormedError.new(line){e.message}
  end
end
