def fu_blksize(st)
  s = st.blksize
  return nil unless s
  return nil if s == 0
  s
end

def fu_stream_blksize(*streams)
  streams.each do |s|
    next unless s.respond_to?(:stat)
    size = fu_blksize(s.stat)
    return size if size
  end
  fu_default_blksize()
end
