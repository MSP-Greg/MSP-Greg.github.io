def self.extend_object(obj)
  super
  obj.instance_eval {@optparse = nil}
end

    #options   (rw)
  
      

      
  
      
    

      Actual ::Gem::OptionParser object, automatically created if nonexistent.
      


      If called with a block, yields the ::Gem::OptionParser object and returns the result of the block. If an ParseError exception occurs in the block, it is rescued, a error message printed to STDERR and nil returned.

def options
  @optparse ||= Gem::OptionParser.new
  @optparse.default_argv = self
  block_given? or return @optparse
  begin
    yield @optparse
  rescue ParseError
    @optparse.warn $!
    nil
  end
end

def options=(opt)
  unless @optparse = opt
    class << self
      undef_method(:options)
      undef_method(:options=)
    end
  end
end

      def getopts(*args)
  ($OPT = ARGV.getopts(*args)).each do |opt, val|
    eval "$OPT_#{opt.gsub(/[^A-Za-z0-9_]/, '_')} = val"
  end
rescue Gem::OptionParser::ParseError
end
