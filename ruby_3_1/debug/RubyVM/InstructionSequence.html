def traceable_lines_norec lines
  code = self.to_a[13]
  line = 0
  code.each{|e|
    case e
    when Integer
      line = e
    when Symbol
      if /\ARUBY_EVENT_/ =~ e.to_s
        lines[line] = [e, *lines[line]]
      end
    end
  }
end

def traceable_lines_rec lines
  self.each_child{|ci| ci.traceable_lines_rec(lines)}
  traceable_lines_norec lines
end
