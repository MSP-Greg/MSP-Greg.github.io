def trap sig, *command, &command_proc
  case sig&.to_sym
  when :INT, :SIGINT
    if defined?(SESSION) && SESSION.active? && SESSION.intercept_trap_sigint?
      return SESSION.save_int_trap(command.empty? ? command_proc : command.first)
    end
  end

  super
end
