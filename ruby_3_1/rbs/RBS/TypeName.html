def initialize(namespace:, name:)
  @namespace = namespace
  @name = name
  @kind = case name.to_s[0,1]
          when /[A-Z]/
            :class
          when /[a-z]/
            :alias
          when "_"
            :interface
          else
            # Defaults to :class
            :class
          end
end

def +(other)
  if other.absolute?
    other
  else
    TypeName.new(
      namespace: self.to_namespace + other.namespace,
      name: other.name
    )
  end
end
