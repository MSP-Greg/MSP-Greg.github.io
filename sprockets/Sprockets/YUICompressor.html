def call(input)
  data = input[:data]

  case input[:content_type]
  when 'application/javascript'
    Autoload::YUI::JavaScriptCompressor.new(@options).compress(data)
  when 'text/css'
    Autoload::YUI::CssCompressor.new(@options).compress(data)
  else
    data
  end
end
