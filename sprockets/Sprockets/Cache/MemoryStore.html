def initialize(max_size = DEFAULT_MAX_SIZE)
  @max_size = max_size
  @cache = {}
  @mutex = Mutex.new
end

def clear(options=nil)
  @mutex.synchronize do
    @cache.clear
  end
  true
end

def get(key)
  @mutex.synchronize do
    exists = true
    value = @cache.delete(key) { exists = false }
    if exists
      @cache[key] = value
    else
      nil
    end
  end
end

def inspect
  @mutex.synchronize do
    "#<#{self.class} size=#{@cache.size}/#{@max_size}>"
  end
end

def set(key, value)
  @mutex.synchronize do
    @cache.delete(key)
    @cache[key] = value
    @cache.shift if @cache.size > @max_size
  end
  value
end
