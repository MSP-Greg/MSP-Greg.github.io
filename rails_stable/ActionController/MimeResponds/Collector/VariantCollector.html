def any(*args, &block)
  if block_given?
    if args.any? && args.none? { |a| a == @variant }
      args.each { |v| @variants[v] = block }
    else
      @variants[:any] = block
    end
  end
end

def variant
  if @variant.empty?
    @variants[:none] || @variants[:any]
  else
    @variants[variant_key]
  end
end

def variant_key
  @variant.find { |variant| @variants.key?(variant) } || :any
end
