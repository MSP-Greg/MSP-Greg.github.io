def body
  @str_body ||= begin
    buf = +""
    each { |chunk| buf << chunk }
    buf
  end
end

def each(&block)
  if @str_body
    return enum_for(:each) unless block_given?

    yield @str_body
  else
    each_chunk(&block)
  end
end

def respond_to?(method, include_private = false)
  if BODY_METHODS.key?(method)
    @buf.respond_to?(method)
  else
    super
  end
end

def to_ary
  if @str_body
    [body]
  else
    @buf = @buf.to_ary
  end
end

def write(string)
  raise IOError, "closed stream" if closed?

  @str_body = nil
  @response.commit!
  @buf.push string
end
