def initialize(separator, matchers)
  @separator = separator
  @matchers  = matchers
  @separator_re = "([^#{separator}]+)"
  super()
end

def visit_OR(node)
  children = node.children.map { |n| visit n }
  "(?:#{children.join(?|)})"
end

def visit_STAR(node)
  re = @matchers[node.left.to_sym]
  re ? "(#{re})" : "(.+)"
end

def visit_SYMBOL(node)
  node = node.to_sym

  return @separator_re unless @matchers.key?(node)

  re = @matchers[node]
  "(#{Regexp.union(re)})"
end
