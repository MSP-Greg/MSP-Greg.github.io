def dedup_scan(regex)
  r = @ss.scan(regex)
  r ? -r : nil
end

def next_token
  return if @ss.eos?

  until token = scan || @ss.eos?; end
  token
end

def scan
  case
    # /
  when @ss.skip(/\//)
    [:SLASH, "/"]
  when @ss.skip(/\(/)
    [:LPAREN, "("]
  when @ss.skip(/\)/)
    [:RPAREN, ")"]
  when @ss.skip(/\|/)
    [:OR, "|"]
  when @ss.skip(/\./)
    [:DOT, "."]
  when text = dedup_scan(/:\w+/)
    [:SYMBOL, text]
  when text = dedup_scan(/\*\w+/)
    [:STAR, text]
  when text = @ss.scan(/(?:[\w%\-~!$&'*,;=@]|\\[:()])/)
    text.tr! "\\", ""
    [:LITERAL, -text]
    # any char
  when text = dedup_scan(/./)
    [:LITERAL, text]
  end
end
