def nary(node, seed)
  last_child = node.children.last
  node.children.inject(seed) { |s, c|
    string = visit(c, s)
    string << "|" unless last_child == c
    string
  }
end

def visit_GROUP(node, seed)
  visit(node.left, seed.dup << "(") << ")"
end
