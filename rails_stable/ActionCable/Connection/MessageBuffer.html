      Overview

def append(message)
  if valid? message
    if processing?
      receive message
    else
      buffer message
    end
  else
    connection.logger.error "Couldn't handle non-string message: #{message.class}"
  end
end

def receive_buffered_messages
  receive buffered_messages.shift until buffered_messages.empty?
end
