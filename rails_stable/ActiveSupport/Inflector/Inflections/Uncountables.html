def add(words)
  words = words.flatten.map(&:downcase)
  concat(words)
  @regex_array += words.map { |word| to_regex(word) }
  self
end

def delete(entry)
  super entry
  @regex_array.delete(to_regex(entry))
end

def to_regex(string)
  /\b#{::Regexp.escape(string)}\Z/i
end

def uncountable?(str)
  @regex_array.any? { |regex| regex.match? str }
end
