def warn(message, *)
  return if SUPPRESSED_WARNINGS.match?(message)

  super

  return unless message.include?(PROJECT_ROOT)
  return if ALLOWED_WARNINGS.match?(message)
  return unless ENV["RAILS_STRICT_WARNINGS"] || ENV["CI"]

  raise message
end
