def parse(data)
  if !data.respond_to?(:read)
    data = StringIO.new(data || "")
  end

  if data.eof?
    {}
  else
    doc = Nokogiri::XML(data)
    raise doc.errors.first if doc.errors.length > 0
    doc.to_hash
  end
end
