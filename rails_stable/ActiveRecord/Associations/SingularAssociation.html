def _create_record(attributes, raise_error = false, &block)
  record = build_record(attributes, &block)
  saved = record.save
  set_new_record(record)
  raise RecordInvalid.new(record) if !saved && raise_error
  record
end

def reader
  ensure_klass_exists!

  if !loaded? || stale_target?
    reload
  end

  target
end
