def defined_for?(name:, expression: nil, validate: nil, **options)
  self.name == name.to_s &&
    (validate.nil? || validate == self.options.fetch(:validate, validate)) &&
    options.all? { |k, v| self.options[k].to_s == v.to_s }
end
