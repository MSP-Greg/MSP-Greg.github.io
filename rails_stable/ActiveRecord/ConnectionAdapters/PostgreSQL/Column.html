def virtual?
  # We assume every generated column is virtual, no matter the concrete type
  @generated.present?
end

def ==(other)
  other.is_a?(Column) &&
    super &&
    identity? == other.identity? &&
    serial? == other.serial?
end

def encode_with(coder)
  coder["serial"] = @serial
  coder["identity"] = @identity
  coder["generated"] = @generated
  super
end
