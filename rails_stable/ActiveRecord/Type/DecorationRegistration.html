def call(registry, *args, **kwargs)
  subtype = registry.lookup(*args, **kwargs.except(*options.keys))
  klass.new(subtype)
end

def matches_options?(**kwargs)
  options.all? do |key, value|
    kwargs[key] == value
  end
end
