def address_list(obj)
  if obj&.respond_to?(:element)
    # Mail 2.8+
    obj.element
  else
    # Mail <= 2.7.x
    obj&.address_list
  end
end

def recipients
  Array(to) + Array(cc) + Array(bcc) + Array(header[:x_original_to]).map(&:to_s) +
    Array(header[:x_forwarded_to]).map(&:to_s)
end
