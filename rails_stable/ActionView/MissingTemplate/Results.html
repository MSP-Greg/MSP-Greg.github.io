def add(path, score)
  if should_record?(score)
    @results << Result.new(path, score)
    @results.sort_by!(&:score)
    @results.pop if @results.size > @size
  end
end

def should_record?(score)
  if @results.size < @size
    true
  else
    score < @results.last.score
  end
end
