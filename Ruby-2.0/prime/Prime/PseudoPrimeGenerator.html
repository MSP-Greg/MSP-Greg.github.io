def each(&block)
  return self.dup unless block
  if @ubound
    last_value = nil
    loop do
      prime = succ
      break last_value if prime > @ubound
      last_value = block.call(prime)
    end
  else
    loop do
      block.call(succ)
    end
  end
end

def with_object(obj)
  return enum_for(:with_object) unless block_given?
  each do |prime|
    yield prime, obj
  end
