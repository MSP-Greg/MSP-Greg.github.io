def element_init
  return if defined?(frameset)
  methods = ""
  # - -
  for element in %w[ FRAMESET ]
    methods << <<-BEGIN + nn_element_def(element) + <<-END
      end
    END
  end

  # - O EMPTY
  for element in %w[ FRAME ]
    methods << <<-BEGIN + nOE_element_def(element) + <<-END
      end
    END
  end
  eval(methods)
end
