      Overview

      Provides an implementation for RSpec::Rails::Matchers#have_http_status matching against Rack symbol http status codes.

      Not intended to be instantiated directly.

      Examples:

      expect(response).to have_http_status(:created)

      See Also:

      • RSpec::Rails::Matchers#have_http_status

def actual_status
  return unless actual

  @actual_status ||= compute_status_from(actual)
end

def compute_status_from(code)
  status, _ = Rack::Utils::SYMBOL_TO_STATUS_CODE.find do |_, c|
    c == code
  end
  status
end

def matches?(response)
  test_response = as_test_response(response)
  @actual = test_response.response_code
  expected == @actual
rescue TypeError => _ignored
  @invalid_response = response
  false
end

def pp_status(status, code)
  if status
    "#{status.inspect} (#{code})"
  else
    code.to_s
  end
end
