def failure_message
  "expected #{describe_inbound_email} to route to #{mailbox}".tap do |msg|
    if receiver
      msg << ", but routed to #{receiver} instead"
    end
  end
end

def matches?(mailbox)
  @mailbox  = mailbox
  @receiver = ApplicationMailbox.router.mailbox_for(inbound_email)

  @receiver == @mailbox
end
