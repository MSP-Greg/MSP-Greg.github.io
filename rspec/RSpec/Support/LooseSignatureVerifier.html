      Overview

def split_args(*args)
  if RSpec::Support.is_a_matcher?(args.last) && @signature.could_contain_kw_args?(args)
    args.pop
    @signature = SignatureWithKeywordArgumentsMatcher.new(@signature)
  end

  super(*args)
end
