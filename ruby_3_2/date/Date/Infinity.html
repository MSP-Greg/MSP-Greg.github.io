def <=>(other)
  case other
  when Infinity; return d <=> other.d
  when Float::INFINITY; return d <=> 1
  when -Float::INFINITY; return d <=> -1
  when Numeric; return d
  else
    begin
      l, r = other.coerce(self)
      return l <=> r
    rescue NoMethodError
    end
  end
  nil
end

def to_f
  return 0 if @d == 0
  if @d > 0
    Float::INFINITY
  else
    -Float::INFINITY
  end
end
