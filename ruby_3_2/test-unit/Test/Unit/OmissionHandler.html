def handle_omitted_error(exception)
  return false unless exception.is_a?(OmittedError)
  omission = Omission.new(name,
                          filter_backtrace(exception.backtrace),
                          exception.message,
                          :method_name => @method_name)
  add_omission(omission)
  true
end
