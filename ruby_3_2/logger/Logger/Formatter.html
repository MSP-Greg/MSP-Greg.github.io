def call(severity, time, progname, msg)
  sprintf(Format, severity, format_datetime(time), Process.pid, severity, progname, msg2str(msg))
end

def msg2str(msg)
  case msg
  when ::String
    msg
  when ::Exception
    "#{ msg.message } (#{ msg.class })\n#{ msg.backtrace.join("\n") if msg.backtrace }"
  else
    msg.inspect
  end
end
