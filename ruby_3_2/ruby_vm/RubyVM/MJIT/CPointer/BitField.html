def *
  byte = Fiddle::Pointer.new(@addr)[0, Fiddle::SIZEOF_CHAR].unpack1('c')
  if @width == 1
    bit = (1 & (byte >> @offset))
    bit == 1
  elsif @width <= 8 && @offset == 0
    bitmask = @width.times.sum { |i| 1 << i }
    byte & bitmask
  else
    raise NotImplementedError.new("not-implemented bit field access: width=#{@width} offset=#{@offset}")
  end
end
