def add_subst!(ty, subst)
  if subst[self]
    subst[self] = subst[self].union(ty)
  else
    subst[self] = ty
  end
  true
end

def substitute(subst, depth)
  if subst[self]
    subst[self].limit_size(depth)
  else
    self
  end
end
