def free_variables(set = Set.new)
  set.tap do
    types.each do |type|
      type.free_variables set
    end
  end
end

def map_type(&block)
  if block
    Tuple.new(
      types: types.map {|type| yield type },
      location: location
    )
  else
    enum_for :map_type
  end
end

def map_type_name(&block)
  Tuple.new(
    types: types.map {|type| type.map_type_name(&block) },
    location: location
  )
end

def sub(s)
  self.class.new(types: types.map {|ty| ty.sub(s) },
                 location: location)
end

def to_s(level = 0)
  if types.empty?
    "[ ]"
  else
    "[ #{types.join(", ")} ]"
  end
end
