def overload?
  case mem = member
  when AST::Members::MethodDefinition
    mem.overload?
  else
    false
  end
end

def ==(other)
  other.is_a?(TypeDef) &&
    other.type == type &&
    other.member == member &&
    other.defined_in == defined_in &&
    other.implemented_in == implemented_in
end
