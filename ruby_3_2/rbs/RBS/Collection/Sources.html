def self.from_config_entry(source_entry)
  case source_entry['type']
  when 'git', nil # git source by default
    __skip__ = Git.new(**source_entry.slice('name', 'revision', 'remote', 'repo_dir').transform_keys(&:to_sym))
  when 'stdlib'
    Stdlib.instance
  when 'rubygems'
    Rubygems.instance
  else
    raise
  end
end
