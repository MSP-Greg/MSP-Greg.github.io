def each_decl
  if block_given?
    members.each do |member|
      if member.is_a?(Base)
        yield(_ = member)
      end
    end
  else
    enum_for :each_decl
  end
end

def each_member
  if block_given?
    members.each do |member|
      if member.is_a?(Members::Base)
        yield(_ = member)
      end
    end
  else
    enum_for :each_member
  end
end
