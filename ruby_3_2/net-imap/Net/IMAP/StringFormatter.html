def string(str)
  str = str.to_s
  if str =~ LITERAL_REGEX
    Literal.new(str)
  else
    QuotedString.new(str)
  end
end

def valid_string?(str)
  str.is_a?(Symbol) || str.respond_to?(:to_str)
end
