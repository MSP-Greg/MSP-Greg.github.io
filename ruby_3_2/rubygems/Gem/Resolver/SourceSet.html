def find_all(req) # :nodoc:
  if set = get_set(req.dependency.name)
    set.find_all req
  else
    []
  end
end

def get_set(name)
  link = @links[name]
  @sets[link] ||= Gem::Source.new(link).dependency_resolver_set if link
end

def prefetch(reqs) # :nodoc:
  reqs.each do |req|
    if set = get_set(req.dependency.name)
      set.prefetch reqs
    end
  end
end
