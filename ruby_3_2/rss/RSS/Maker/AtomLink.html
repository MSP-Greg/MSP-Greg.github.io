def to_feed(feed, current)
  link = feed.class::Link.new
  set = setup_values(link)
  if set
    current.links << link
    set_parent(link, current)
    setup_other_elements(feed)
  else
    raise NotSetError.new(self.class.not_set_name,
                          not_set_required_variables)
  end
end
