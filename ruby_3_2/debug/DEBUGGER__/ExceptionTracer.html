def setup
  @tracer = TracePoint.new(:raise) do |tp|
    next if skip?(tp)

    exc = tp.raised_exception

    out tp, " #{colorize_magenta(exc.inspect)}"
  rescue Exception => e
    p e
  end
end

def skip_with_pattern?(tp)
  super && !tp.raised_exception.inspect.match?(@pattern)
end
