def self.make(*args)
  result = empty
  args.reverse_each do |item|
    result = cons(item, result)
  end
  result
end

def ==(other)
  current = self
  while ! current.empty? && ! other.empty?
    return false if current.head != other.head
    current = current.tail
    other = other.tail
  end
  current.empty? && other.empty?
end

def each
  current = self
  while ! current.empty?
    yield(current.head)
    current = current.tail
  end
  self
end
