def call(names, element)
  return names if names.size == 1
  str = normalize element
  return [str] if names.include? str
  checker = ::DidYouMean::SpellChecker.new(dictionary: names)
  checker.correct(str)
end

def normalize(leaf)
  str = leaf.dup
  str.downcase!
  return str unless str.include? '@'
  str.tr!('@', '  ')
end
