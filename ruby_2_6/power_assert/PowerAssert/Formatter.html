def inspect
  if PowerAssert.configuration._colorize_message
    if PowerAssert.configuration._use_pp
      width = [Pry::Terminal.width! - 1 - @indent, 10].max
      Pry::ColorPrinter.pp(@value, '', width)
    else
      Pry::Code.new(@value.inspect).highlighted
    end
  else
    if PowerAssert.configuration._use_pp
      PP.pp(@value, '')
    else
      @value.inspect
    end
  end
end
