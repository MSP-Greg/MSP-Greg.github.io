def format_time(time)
  if time.utc?
    time.strftime("%Y-%m-%d %H:%M:%S.%9N Z")
  else
    time.strftime("%Y-%m-%d %H:%M:%S.%9N %:z")
  end
end

def visit_String(str)
  return super unless str == '=' # or whatever you want

  quote = Psych::Nodes::Scalar::SINGLE_QUOTED
  @emitter.scalar str, nil, nil, false, true, quote
end
