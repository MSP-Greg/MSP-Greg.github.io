def self._valid_method?(method)
  iseq = RubyVM::InstructionSequence.compile("().#{method}", nil, nil, 0, false)
rescue SyntaxError
  false
else
  iseq.to_a.dig(-1, 1, 1, :mid) == method.to_sym
end
