def <=>(other)
  return unless other.is_a?(Identifier)
  [name, version, platform_string] <=> [other.name, other.version, other.platform_string]
end

def platform_string
  platform_string = platform.to_s
  platform_string == Index::RUBY ? Index::NULL : platform_string
end
