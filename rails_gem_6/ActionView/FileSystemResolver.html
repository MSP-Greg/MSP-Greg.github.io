def initialize(path)
  raise ArgumentError, "path already is a Resolver class" if path.is_a?(Resolver)
  super()
  @path = File.expand_path(path)
end

def eql?(resolver)
  self.class.equal?(resolver.class) && to_path == resolver.to_path
end
