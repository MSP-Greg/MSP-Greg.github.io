def error_wrapping(html_tag)
  if object_has_errors?
    Base.field_error_proc.call(html_tag, self)
  else
    html_tag
  end
end

def object
  @active_model_object ||= begin
    object = super
    object.respond_to?(:to_model) ? object.to_model : object
  end
end
