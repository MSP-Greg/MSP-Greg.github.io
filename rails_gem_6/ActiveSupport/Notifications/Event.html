def initialize(name, start, ending, transaction_id, payload)
  @name           = name
  @payload        = payload.dup
  @time           = start
  @transaction_id = transaction_id
  @end            = ending
  @children       = []
  @cpu_time_start = 0
  @cpu_time_finish = 0
  @allocation_count_start = 0
  @allocation_count_finish = 0
end
