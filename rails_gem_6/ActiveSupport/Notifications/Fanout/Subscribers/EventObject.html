def finish(name, id, payload)
  stack = Thread.current[:_event_stack]
  event = stack.pop
  event.payload = payload
  event.finish!
  @delegate.call event
end

def start(name, id, payload)
  stack = Thread.current[:_event_stack] ||= []
  event = build_event name, id, payload
  event.start!
  stack.push event
end
