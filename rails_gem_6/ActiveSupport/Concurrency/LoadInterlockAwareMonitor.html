def mon_enter
  mon_try_enter ||
    ActiveSupport::Dependencies.interlock.permit_concurrent_loads { super }
end

def synchronize
  Thread.handle_interrupt(EXCEPTION_NEVER) do
    mon_enter

    begin
      Thread.handle_interrupt(EXCEPTION_IMMEDIATE) do
        yield
      end
    ensure
      mon_exit
    end
  end
end
