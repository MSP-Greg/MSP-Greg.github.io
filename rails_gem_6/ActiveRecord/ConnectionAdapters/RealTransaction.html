def commit
  connection.commit_db_transaction if materialized?
  @state.full_commit!
end

def materialize!
  if isolation_level
    connection.begin_isolated_db_transaction(isolation_level)
  else
    connection.begin_db_transaction
  end

  super
end

def rollback
