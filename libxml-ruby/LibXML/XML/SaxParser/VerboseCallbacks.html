def on_start_element_ns(name, attributes, prefix, uri, namespaces)
  STDOUT << "on_start_element_ns" << "\n" <<
            "  name: " << name << "\n" <<
            "  attr: " << (attributes || Hash.new).inspect << "\n" <<
            "  prefix: " << prefix << "\n" <<
            "  uri: " << uri << "\n" <<
            "  ns_defs: " << (namespaces || Hash.new).inspect << "\n"
  STDOUT.flush
end
