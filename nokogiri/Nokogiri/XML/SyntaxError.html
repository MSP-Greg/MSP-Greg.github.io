def level_to_s
  case level
  when 3 then "FATAL"
  when 2 then "ERROR"
  when 1 then "WARNING"
  end
end

def location_to_s
  return if nil_or_zero?(line) && nil_or_zero?(column)

  "#{line}:#{column}"
end

def to_s
  message = super.chomp
  [location_to_s, level_to_s, message]
    .compact.join(": ")
    .force_encoding(message.encoding)
end
