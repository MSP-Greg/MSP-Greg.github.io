def self.included(base)
  base.class_eval do
    alias_method :_safe?, :safe?
    alias_method :_idempotent?, :idempotent?

    def safe?
      _safe? or propfind?
    end

    def idempotent?
      _idempotent? or propfind? or move? or unlock? # or lock?
    end
  end
end
