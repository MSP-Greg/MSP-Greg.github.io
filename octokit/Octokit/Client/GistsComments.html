def create_gist_comment(gist_id, body, options = {})
  opts = options.dup
  opts[:body] = body
  post "gists/#{Gist.new gist_id}/comments", opts
end

def update_gist_comment(gist_id, comment_id, body, options = {})
  opts = options.dup
  opts[:body] = body
  patch "gists/#{Gist.new gist_id}/comments/#{comment_id}", opts
end
