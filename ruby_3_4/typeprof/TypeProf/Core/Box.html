def initialize(node)
  @node = node
  @changes = ChangeSet.new(node, self)
  @destroyed = false
  $box_counts[Box] += 1
  $box_counts[self.class] += 1
end

def destroy(genv)
  $box_counts[self.class] -= 1
  $box_counts[Box] -= 1
  @destroyed = true
  @changes.reinstall(genv) # rollback all changes
end

def run(genv)
  return if @destroyed
  run0(genv, @changes)
  @changes.reinstall(genv)
end
