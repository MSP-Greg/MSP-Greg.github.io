def self.[](*elems)
  h = Hash.new(false)
  elems.each {|elem| h[elem] = true }
  new(h)
end

def pretty_print(q)
  q.text "Set["
  q.group do
    q.nest(1) do
      @hash.each_key do |elem|
        q.breakable ""
        q.pp elem
        q.text ","
      end
    end
    q.breakable ""
  end
  q.text "]"
end
