def get_self(genv)
  case @scope_level
  when :instance
    mod = genv.resolve_cpath(@cpath || [])
    type_params = mod.type_params.map {|ty_param| Source.new() } # TODO: better support
    ty = Type::Instance.new(genv, mod, type_params)
    Source.new(ty)
  when :class
    Source.new(Type::Singleton.new(genv, genv.resolve_cpath(@cpath || [])))
  else
    Source.new()
  end
end
