def to_lsp
  json = {
    range: code_range.to_lsp,
    source: "TypeProf",
    message: @msg,
  }
  json[:severity] = SEVERITY[@severity] if @severity
  json[:tags] = @tags.map {|tag| TAG[tag] } if @tags
  json
end
