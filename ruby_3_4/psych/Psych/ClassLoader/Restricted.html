def find klassname
  if @classes.include? klassname
    super
  else
    raise DisallowedClass.new('load', klassname)
  end

def symbolize sym
  return super if @symbols.empty?

  if @symbols.include? sym
    super
  else
    raise DisallowedClass.new('load', 'Symbol')
  end
end
