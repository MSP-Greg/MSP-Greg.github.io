def Hi(str, salt, iterations)
  length = digest.digest_length
  OpenSSL::KDF.pbkdf2_hmac(
    str,
    salt:       salt,
    iterations: iterations,
    length: length,
    hash: digest,
  )
end

def XOR(str1, str2)
  str1.unpack("C*")
    .zip(str2.unpack("C*"))
    .map {|a, b| a ^ b }
    .pack("C*")
end
