def process_flags flags
  flags.reject { |flag| # all hits are truthy, so this works out well
    case flag
    when /^-I(.*)/ then
      rb_flags << flag
    when /^-d/ then
      rb_flags << flag
    when /^-w/ then
      rb_flags << flag
    else
      false
    end
  }
end
