def self.decode(msg) # :nodoc:
  addresses = msg.get_list { IPv6.new(msg.get_bytes(16)) }
  return self.new(addresses)
end

def encode(msg) # :nodoc:
  @addresses.each do |address|
    msg.put_bytes(address.address)
  end
