def auth(user, secret)
  token = xoauth2_string(user, secret)

  finish("AUTH XOAUTH2 #{base64_encode(token)}")
end

def xoauth2_string(user, secret)
  "user=#{user}\1auth=Bearer #{secret}\1\1"
end
