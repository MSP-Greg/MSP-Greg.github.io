      Overview

      An Array wrapper that can be sent to another server via ::DRb.

      All entries in the array will be dumped or be references that point to the local server.

def initialize(ary)
  @ary = ary.collect { |obj|
    if obj.kind_of? DRbUndumped
      DRbObject.new(obj)
    else
      begin
        Marshal.dump(obj)
        obj
      rescue
        DRbObject.new(obj)
      end
    end
  }
end
