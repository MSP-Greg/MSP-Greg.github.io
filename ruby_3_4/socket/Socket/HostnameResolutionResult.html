def initialize(size)
  @size = size
  @taken_count = 0
  @rpipe, @wpipe = IO.pipe
  @results = []
  @mutex = Mutex.new
end

def add(family, result)
  @mutex.synchronize do
    @results.push [family, result]
    @wpipe.putc HOSTNAME_RESOLUTION_QUEUE_UPDATED
  end
end

def get
  return nil if @results.empty?

  res = nil

  @mutex.synchronize do
    @rpipe.getbyte
    res = @results.shift
  end

  @taken_count += 1
  close if @taken_count == @size
  res
end
