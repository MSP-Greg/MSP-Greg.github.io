  @callback || ->(_, _, message) { warn(message) }
end

      
  
      

      

      


      
  
      
    .callback=(value)   (rw)
  
      

      
  
      
    

      Set the custom deprecation callback. By default this issues a deprecation warning.

def call(klass, method, message)
  callback && callback.call(klass, method, message)
end
