def do_GET(request, response) # rubocop:disable Style/MethodName
  status, headers, body = *adapter.router.call(request)
  response.status = status
  response.body = body.is_a?(Array) ? body[0] : body
  headers.each do |key, value|
    response[key] = value
  end
end
