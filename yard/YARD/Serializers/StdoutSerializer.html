def serialize(_object, data)
  print(@wrap ? word_wrap(data, @wrap) : data)
end

def word_wrap(text, _length = 80)
  # See ruby-talk/10655 / Ernest Ellingson
  text.gsub(/\t/, "     ").gsub(/.{1,50}(?:\s|\Z)/) do
    ($& + 5.chr).gsub(/\n\005/, "\n").gsub(/\005/, "\n")
  end
end
