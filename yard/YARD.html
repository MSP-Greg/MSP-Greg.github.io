def self.windows?
  return @windows if defined? @windows
  require 'rbconfig'
  @windows =
    ::RbConfig::CONFIG['host_os'] =~ /mingw|win32|cygwin/ ? true : false
ensure
  @windows ||= false
end

      Deprecated.

      Loads gems that match the name 'yard-' (recommended) or 'yard_' except

after YARD is loaded to allow plugin support.
