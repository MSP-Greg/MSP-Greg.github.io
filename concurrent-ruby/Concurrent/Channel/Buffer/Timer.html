def do_poll
  synchronize do
    if ns_closed?
      return Concurrent::NULL, false
    elsif Concurrent.monotonic_time >= @tick
      # only one listener gets notified
      self.closed = true
      return Concurrent::Channel::Tick.new(@tick), false
    else
      return nil, true
    end
  end
end

def next
  loop do
    tick, more = do_poll
    return tick, more if tick
    Thread.pass
  end
end

def poll
  tick, _ = do_poll
  tick = Concurrent::NULL unless tick
  tick
end

def take
  loop do
    tick, _ = do_poll
    if tick
      return tick
    else
      Thread.pass
    end
  end
end
