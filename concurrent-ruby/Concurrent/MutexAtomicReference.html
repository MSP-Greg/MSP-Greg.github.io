def _compare_and_set(old_value, new_value)
  synchronize do
    if @value.equal? old_value
      @value = new_value
      true
    else
      false
    end
  end
end

def get_and_set(new_value)
  synchronize do
    old_value = @value
    @value = new_value
    old_value
  end
end

def synchronize
  if @Lock.owned?
    yield
  else
    @Lock.synchronize { yield }
  end
end
