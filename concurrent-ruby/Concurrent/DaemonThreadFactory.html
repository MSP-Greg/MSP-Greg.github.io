def initialize(daemonize = true)
  @daemonize = daemonize
  @java_thread_factory = java.util.concurrent.Executors.defaultThreadFactory

def newThread(runnable)
  thread = @java_thread_factory.newThread(runnable)
  thread.setDaemon(@daemonize)
  return thread
end
