def select(*channels)
  # TODO (pitr-ch 26-Mar-2016): re-do, has to be non-blocking
  future do
    # noinspection RubyArgCount
    Channel.select do |s|
      channels.each do |ch|
        s.take(ch) { |value| [value, ch] }
      end
    end
  end
end
