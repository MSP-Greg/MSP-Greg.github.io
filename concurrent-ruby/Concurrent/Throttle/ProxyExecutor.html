def inner_post(*arguments, &task)
  @Executor.post(*arguments) do |*args|
    begin
      task.call(*args)
    ensure
      @Throttle.release
    end
  end
end

def post(*args, &task)
  if (event = @Throttle.acquire_or_event)
    event.on_resolution! { inner_post(*args, &task) }
  else
    inner_post(*args, &task)
  end
end
