def initialize(message, future, sender, address)
  @message = message
  @future  = Type! future, Promises::ResolvableFuture, NilClass
  @sender  = Type! sender, Reference, Thread
  @address = Type! address, Reference
end

def sender_path
  if sender.is_a? Reference
    sender.path
  else
    sender.to_s
  end
end
