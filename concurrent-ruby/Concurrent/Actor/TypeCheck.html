def self.error(value, message, types)
  raise TypeError,
        "Value (#{value.class}) '#{value}' #{message} any of: #{types.join('; ')}."
end

def Child!(value, *types)
  Child?(value, *types) or
      TypeCheck.error(value, 'is not child', types)
  value
end

def Child?(value, *types)
  Type?(value, Class) &&
      types.any? { |t| value <= t }
end
