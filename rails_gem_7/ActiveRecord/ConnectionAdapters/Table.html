      Overview

def column(column_name, type, index: nil, **options)
  raise_on_if_exist_options(options)
  @base.add_column(name, column_name, type, **options)
  if index
    index_options = index.is_a?(Hash) ? index : {}
    index(column_name, **index_options)
  end
end
