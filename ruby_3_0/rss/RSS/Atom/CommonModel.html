def self.append_features(klass)
  super
  klass.install_must_call_validator("atom", URI)
  [
   ["lang", :xml],
   ["base", :xml],
  ].each do |name, uri, required|
    klass.install_get_attribute(name, uri, required, [nil, :inherit])
  end
  klass.class_eval do
    class << self
      # Returns the Atom URI W3C Namespace
      def required_uri
        URI
      end

      # Returns true
      def need_parent?
        true
      end
    end
  end
end
