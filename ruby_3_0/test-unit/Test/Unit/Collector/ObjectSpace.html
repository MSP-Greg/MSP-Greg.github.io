def collect(name=NAME)
  suite = TestSuite.new(name)
  sub_suites = []
  @source.each_object(Class) do |klass|
    if(Test::Unit::TestCase > klass)
      add_suite(sub_suites, klass.suite)
    end
  end
  sort(sub_suites).each{|s| suite << s}
  suite
end
