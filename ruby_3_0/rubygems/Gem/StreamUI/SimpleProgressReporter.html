def initialize(out_stream, size, initial_message,
               terminal_message = "complete")
  @out = out_stream
  @total = size
  @count = 0
  @terminal_message = terminal_message

  @out.puts initial_message
end

def updated(message)
  @count += 1
  @out.print "."
  @out.flush
end
