def read(len = nil)
  if len
    s = super(len, String.new, true)
    return s.empty? ? nil : s
  else
    result = String.new
    while s = super(DEFAULT_BLOCKSIZE, String.new, true)
      break if s.empty?
      result << s
    end
    return result
  end
end

def readline
  line = gets
  if line.nil?
    raise EOFError, "end of file reached"
  end
  return line
end
