def matches?(window)
  @window = window
  @wait_time = Capybara::Queries::BaseQuery.wait(@options, window.session.config.default_max_wait_time)
  timer = Capybara::Helpers.timer(expire_in: @wait_time)
  while window.exists?
    return false if timer.expired?

    sleep 0.01
  end
  true
end
