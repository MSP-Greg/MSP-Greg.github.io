def assert_valid_keys
  invalid_options = @options.keys & COUNT_KEYS
  unless invalid_options.empty?
    raise ArgumentError, "Match queries don't support quantity options. Invalid keys - #{invalid_options.join(', ')}"
  end

  super
end
