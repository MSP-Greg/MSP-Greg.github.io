def run
  @params => {
    textDocument: { uri: },
    position: pos,
  }
  text = @server.open_texts[uri]
  unless text
    respond(nil)
    return
  end
  str = @server.hover(text.path, TypeProf::CodePosition.from_lsp(pos))
  if str
    respond(contents: { language: "ruby", value: str })
  else
    respond(nil)
  end
end
