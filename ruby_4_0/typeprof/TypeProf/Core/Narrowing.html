def and(other)
  new_map = @map.dup
  other.map.each do |var, constraint|
    new_map[var] = new_map[var] ? new_map[var].and(constraint) : constraint
  end
  Narrowing.new(new_map)
end

def or(other)
  new_map = {}
  @map.each do |var, constraint|
    new_map[var] = constraint.or(other.map[var]) if other.map[var]
  end
  Narrowing.new(new_map)
end
