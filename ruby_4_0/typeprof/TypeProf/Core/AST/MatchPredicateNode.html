def initialize(raw_node, lenv)
  super(raw_node, lenv)
  @value = AST.create_node(raw_node.value, lenv)
  @pat = AST.create_pattern_node(raw_node.pattern, lenv)
end

  @pat.install(genv)
  Source.new(genv.true_type, genv.false_type)
end
