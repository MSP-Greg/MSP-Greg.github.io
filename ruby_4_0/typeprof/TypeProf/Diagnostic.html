def to_lsp(severity: :error)
  json = {
    range: code_range.to_lsp,
    source: "TypeProf",
    message: @msg,
  }
  json[:severity] = SEVERITY[severity]
  json[:tags] = @tags.map {|tag| TAG[tag] } if @tags
  json
end
