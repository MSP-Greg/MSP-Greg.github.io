      Overview

def on_heredoc_dedent(val, width)
  val.map! do |e|
    next e if Symbol === e and /_content\z/ =~ e
    if Array === e and e[0] == :@tstring_content
      e = dedent_element(e, width)
    elsif String === e
      dedent_string(e, width)
    end
    e
  end
  val
end
