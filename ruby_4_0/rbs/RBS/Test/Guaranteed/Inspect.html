def inspect
  string = "<#{self.class.name}:"

  instance_variables.each_with_index do |variable, index|
    string.concat ', ' unless index.zero?
    string.concat "#{variable}: #{guaranteed_inspect(instance_variable_get(variable))}"
  end

  string.concat '>'
end
