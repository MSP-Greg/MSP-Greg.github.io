def map_type(&block)
  if block
    ClassInstance.new(
      name: name,
      args: args.map {|type| yield type },
      location: location
    )
  else
    enum_for :map_type
  end
end

def map_type_name(&block)
  ClassInstance.new(
    name: yield(name, location, self),
    args: args.map {|type| type.map_type_name(&block) },
    location: location
  )
end

def sub(s)
  return self if s.empty?

  self.class.new(name: name,
                 args: args.map {|ty| ty.sub(s) },
                 location: location)
end
