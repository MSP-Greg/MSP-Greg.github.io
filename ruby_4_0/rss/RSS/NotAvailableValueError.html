def initialize(tag, value, attribute=nil)
  @tag, @value, @attribute = tag, value, attribute
  message = "value <#{value}> of "
  message << "attribute <#{attribute}> of " if attribute
  message << "tag <#{tag}> is not available."
  super(message)
end
