def compile
  k = @tab.keys
  k.sort!
  k.reverse!
  k.collect!{|path| Regexp.escape(path) }
  @scanner = Regexp.new("\\A(" + k.join("|") +")(?=/|\\z)")
end

def delete(dir)
  dir = normalize(dir)
  res = @tab.delete(dir)
  compile
  res
end

def normalize(dir)
  ret = dir ? dir.dup : ""
  ret.sub!(%r|/+\z|, "")
  ret
end
