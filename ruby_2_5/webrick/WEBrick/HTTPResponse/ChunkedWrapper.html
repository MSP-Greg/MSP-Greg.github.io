def write(buf)
  return 0 if buf.empty?
  socket = @socket
  @resp.instance_eval {
    size = buf.bytesize
    data = "#{size.to_s(16)}#{CRLF}#{buf}#{CRLF}"
    socket.write(data)
    data.clear
    @sent_size += size
    size
  }
end
