def sender(msg, data, host, port=Port)
  lazy_initialize
  id = DNS.allocate_request_id(host, port)
  request = msg.encode
  request[0,2] = [id].pack('n')
  sock = @socks_hash[host.index(':') ? "::" : "0.0.0.0"]
  return @senders[id] =
    UnconnectedUDP::Sender.new(request, data, sock, host, port)
end
