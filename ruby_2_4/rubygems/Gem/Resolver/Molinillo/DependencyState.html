def pop_possibility_state
  PossibilityState.new(
    name,
    requirements.dup,
    activated,
    requirement,
    [possibilities.pop],
    depth + 1,
    conflicts.dup
  ).tap do |state|
    state.activated.tag(state)
  end
end
